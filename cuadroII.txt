function roundTo3(num,pow){
 
 if( 
 isNaN( num ) ){
 num=Int(num);
 num = num.toLocaleString(['ban', 'id']);
 num = num.toFixed(0); 

 num = 0;
 
 }
 
 num *= Math.pow(10,pow);
 num = (Math.round(num)/Math.pow(10,pow))+ "" ;
 if(num.indexOf("") == -1)
 num += "" ;
 while(num.length - num.indexOf(".") - 1 < pow)
 num += "0" ;

 return num;
}
